# Norm Z function declaration
normalisasi=function(x){
  xm=colMeans(x)  
  jb=dim(x)[1] 
  xc=x-rep(xm,each=jb) 
  sdx=apply(x,2,sd) 
  xstd=xc/rep(sdx,each=jb)
  output=data.frame(xstd)
  #output=data.frame(xstd,x)
  return(output)
}


# Min-Max Norm function declaration
min_max_norm <- function(x) {
    (x - min(x)) / (max(x) - min(x))
  }
